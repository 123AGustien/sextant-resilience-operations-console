/** * Sextant Intelligence v11 * Natural language → simulation control layer */ class SextantIntelligence { constructor(orchestra, reasoning) { this.orchestra = orchestra; this.reasoning = reasoning; } /** * Main AI entry point */ ask(question) { if (!question || typeof question !== "string") { return { status: "EMPTY_INPUT", message: "No valid question provided" }; } const q = question.toLowerCase().trim(); // ------------------------- // SAFETY CHECKS // ------------------------- if (!this.orchestra) { return { status: "NO_ORCHESTRA", message: "Orchestra layer missing" }; } // ------------------------- // SIMULATION COMMANDS // ------------------------- if (q.includes("run") || q.includes("simulate")) { const frame = this.orchestra.runStep("normal"); return { intent: "SIMULATION_RUN", frame }; } if (q.includes("failure")) { const frame = this.orchestra.runStep("failure"); return { intent: "FAILURE_SIMULATION", frame }; } if (q.includes("cascade")) { const frame = this.orchestra.runStep("cascade"); return { intent: "CASCADE_SIMULATION", frame }; } // ------------------------- // RESET COMMAND // ------------------------- if (q.includes("reset")) { this.orchestra.reset(); return { intent: "RESET", message: "System reset completed" }; } // ------------------------- // REASONING LAYER // ------------------------- if (q.includes("why") || q.includes("explain")) { if (!this.reasoning) { return { status: "NO_REASONING", message: "Reasoning layer missing" }; } const explanation = this.reasoning.explainLatest(); return { intent: "REASONING_QUERY", explanation }; } // ------------------------- // STATUS QUERY // ------------------------- if (q.includes("status") || q.includes("state")) { const latest = this.orchestra.getLatest?.(); return { intent: "STATUS_QUERY", state: latest?.state || "NO_DATA", frame: latest || null }; } // ------------------------- // DEFAULT RESPONSE // ------------------------- const latest = this.orchestra.getLatest?.(); return { intent: "GENERAL_QUERY", message: "No mapping found. Try: run, failure, cascade, explain, reset, status", latestFrame: latest || null }; } } // expose globally window.SextantIntelligence = SextantIntelligence;
